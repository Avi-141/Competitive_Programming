ull mod_mul(ull a, ull b, ull &c) {
		ull x = a * (b & (po - 1)) % c;
			while ((b >>= bits) > 0) {
						a = (a << bits) % c;
								x += (a * (b & (po - 1))) % c;
									}
				return x % c;
}
